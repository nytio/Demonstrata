\ifPDFTeX
  \newcommand{\leanmonofont}{\ttfamily}
\else
  % Lean source frequently uses Unicode-only syntax such as strict implicit
  % binders `⦃ ⦄` and generated names containing `✝`. Prefer a mono font with
  % broader coverage for the appendix, while keeping a safe fallback.
  \IfFontExistsTF{FreeMono}
    {%
      \setmonofont{FreeMono}[Scale=MatchLowercase]%
      \newfontfamily\leanmonofont{FreeMono}[Scale=MatchLowercase]%
    }
    {%
      \setmonofont{Noto Sans Mono}[Scale=MatchLowercase]%
      \newfontfamily\leanmonofont{Noto Sans Mono}[Scale=MatchLowercase]%
      % Fallbacks for Lean-specific symbols that are not covered by some mono
      % fonts. Keep the mathematical meaning even when the exact glyph is not
      % available.
      \newunicodechar{⦃}{\ensuremath{\lbrace\mkern-3mu\lbrace}}
      \newunicodechar{⦄}{\ensuremath{\rbrace\mkern-3mu\rbrace}}
      \newunicodechar{✝}{\ensuremath{\dagger}}
    }
\fi

\ExplSyntaxOn
\NewDocumentCommand{\leanname}{m}{\texttt{#1}}
\NewDocumentCommand{\leaninline}{m}
  {
    \ifmmode
      \text{\leanname{#1}}
    \else
      \leanname{#1}
    \fi
  }
\NewDocumentCommand{\leanlink}{m m}
  {
    \ifmmode
      \text{\hyperlink{#1}{\leanname{#2}}}
    \else
      \hyperlink{#1}{\leanname{#2}}
    \fi
  }
\cs_new:Npn \lean_lookup:n #1
  {
    \cs_if_exist:cTF { leanref@#1 }
      { \use:c { leanref@#1 } }
      { \leaninline{#1} }
  }
\cs_new_protected:Npn \lean_render:n #1
  {
    \seq_set_split:Nnn \l_tmpa_seq { , } { #1 }
    \seq_map_indexed_inline:Nn \l_tmpa_seq
      {
        \int_compare:nNnT { ##1 } > { 1 } { ,\allowbreak\space }
        \lean_lookup:n { \tl_trim_spaces:n { ##2 } }
      }
  }
